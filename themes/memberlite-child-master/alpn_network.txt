function pte_handle_table_filter( $json ) {   //add stuff to wpDataTables return data

//  alpn_log("Handling pte_handle_table_filter...");
  $qVars = $_GET;
  $tableId = isset($qVars['table_id']) ? $qVars['table_id'] : 0;
  $action = isset($qVars['action']) ? $qVars['action'] : '';

  if ($tableId == 3 && $action == 'get_wdtable'){   //Getting topic link details
    global $wpdb;
    $allLinks = $topics = array();
    $tableData = json_decode($json, true);
    $data = isset($tableData['data']) ? $tableData['data'] : '';
    foreach ($data as $key => $value) {
      $topics[] = $value[0];
    }
    if ($topics) {  //get all linked topics and meta for displayed topics. This makes sure that links are current. We will still check for permission.
      $allTopics = implode(',', $topics);
      $query = "
            SELECT l.owner_topic_id_1 AS owner_topic_id, l.owner_topic_id_2 AS connected_topic_id, t.name, t.owner_id, t.dom_id, o.name AS owner_name, o.topic_content AS owner_content, o.image_handle AS owner_icon FROM `alpn_topic_links` l LEFT JOIN `alpn_topics` t ON t.id = l.owner_topic_id_2 LEFT JOIN `alpn_topics` o ON o.owner_id = t.owner_id AND o.topic_type_id = 5 WHERE l.owner_topic_id_1 IN ({$allTopics})
            UNION
            SELECT l.owner_topic_id_2 AS owner_topic_id, l.owner_topic_id_2 AS connected_topic_id, t.name, t.owner_id, t.dom_id, o.name AS owner_name, o.topic_content AS owner_content, o.image_handle AS owner_icon FROM `alpn_topic_links` l LEFT JOIN `alpn_topics` t ON t.id = l.owner_topic_id_1 LEFT JOIN `alpn_topics` o ON o.owner_id = t.owner_id AND o.topic_type_id = 5 WHERE l.owner_topic_id_2 IN ({$allTopics}) ORDER BY name ASC
      ";
      $results = $wpdb->get_results(
        $wpdb->prepare($query)
       );
       $allLinks = array();
       foreach($results as $key => $value) {
         $topicContent = json_decode($value->owner_content, true);
         $value->owner_first = $topicContent['person_givenname'];   //TODO change to json extract in query but dots in names still causing problems. I should have used underscores
         unset($value->owner_content);
         $allLinks[] = (array)$value;
       }
       foreach ($data as $key => $value) { //regroup topic links by topic and pass to client.
         $topic = $value[0];
         $newLinks = array();
         foreach ($allLinks as $linkData) {
           if ($linkData['owner_topic_id'] == $topic){
              $newLinks[] = $linkData;
            }
         }
         $data[$key][10] = json_encode($newLinks);  //for wpdatatable, position is everything. TODO this is currently using a non-virtual field in alpn_topics as a placeholder. Make virtual or add dynamically.
       }
       $tableData['data'] = $data;
       //alpn_log($tableData);
       $json = json_encode($tableData);
    }
  }
  return $json;
}
